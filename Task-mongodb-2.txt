Find all the topics and tasks which are thought in the month of October

db.tasks.aggregate([
        {
            $lookup: {
                from: "topics",
                localField: "taskid",
                foreignField: "topicid",
                as: "topicInfo"
            }
        },
          { $match:{"date": { $gte:  "2020-10-01" , $lte:  "2020-10-31"  }}},
          { $group:{_id:{ taskid:"$taskid", date:"$date", task:"$task", studentname:"$student name",topics:"$topicInfo.topics" }}},
          { $project:{"topicInfo.topics":1}},
          { $sort:{_id: 1}}
          
  ]);
